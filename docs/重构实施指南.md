#### 标准模板

```javascript
const BaseTtsService = require('../core/BaseTtsService');
const TtsException = require('../core/TtsException');
const { audioStorageManager } = require('../../../shared/utils/audioStorage');

class XxxTtsService extends BaseTtsService {
  constructor(config = {}) {
    super({
      provider: 'xxx',
      serviceType: 'xxx',
      ...config
    });
  }

  /**
   * 文本转语音
   * @param {string} text - 要转换的文本
   * @param {Object} options - 转换选项
   * @returns {Promise<Object>} 转换结果
   */
  async synthesize(text, options = {}) {
    try {
      // 验证输入
      this.validateText(text);
      this.validateOptions(options);

      // 调用服务商API
      const audioBuffer = await this.callProviderAPI(text, options);

      // 保存音频文件（使用 audioStorageManager）
      const audioFile = await this.audioStorage.saveAudioFile(audioBuffer, {
        extension: options.format || 'mp3',
        metadata: {
          provider: this.provider,
          serviceType: this.serviceType,
          text: text.substring(0, 50)
        }
      });

      return {
        text: text,
        audioUrl: audioFile.url,
        provider: this.provider,
        serviceType: this.serviceType
      };

    } catch (error) {
      throw TtsException.SynthesisFailed(
        `${this.provider} ${this.serviceType} synthesis failed: ${error.message}`,
        { originalError: error.message }
      );
    }
  }

  /**
   * 调用服务商API（子类实现）
   */
  async callProviderAPI(text, options) {
    // 实现具体的API调用逻辑
  }

  /**
   * 获取可用音色列表
   */
  getAvailableVoices() {
    // 返回音色列表
  }
}

// 创建单例

module.exports = xxxTtsService;
```

#### 修改：UnifiedTtsController.js


**简化职责：**
```javascript
// Controller 只负责：
// 1. HTTP 协议转换（req → 参数）
// 2. 响应封装（结果 → res）
// 3. 错误处理（异常 → HTTP状态码）

// 业务逻辑交给 ServiceManager
async synthesize(req, res) {
  try {
    const { service, text, systemId, ...options } = req.body;

    // 如果有 systemId，转换为提供商参数
    if (systemId) {
      const params = voiceModelMapper.systemIdToProviderParams(systemId);
      return await this.serviceManager.synthesize(
        params.provider,
        params.serviceType,
        text,
        { ...params, ...options }
      );
    }

    // 否则使用传统的 service 参数
    const { provider, serviceType } = this.parseServiceIdentifier(service);
    const result = await this.serviceManager.synthesize(provider, serviceType, text, options);

    res.json({ success: true, data: result });
  } catch (error) {
    this.handleError(error, res);
  }
}
```

## 重构收益

### 代码质量
- ✅ 统一的异常处理（TtsException）
- ✅ 统一的服务接口（synthesize）
- ✅ 统一的导出方式（类+单例）
- ✅ 删除了冗余的缓存逻辑
- ✅ 清晰的职责划分

### 可维护性
- ✅ 配置驱动，易于添加新音色
- ✅ 向后兼容，不破坏现有功能
- ✅ 代码结构清晰，易于理解
- ✅ 删除重复代码，遵循DRY原则

### 性能
- ⚠️ 删除缓存可能略微增加响应时间
- ✅ 减少磁盘空间占用
- ✅ 避免缓存一致性问题

## 风险和注意事项

### 潜在风险
1. **删除缓存的影响**
   - 相同文本每次都调用API
   - 响应时间增加
   - API调用次数增加

2. **兼容性问题**
   - 旧客户端可能直接传服务商ID
   - 需要保持双模式支持

### 缓解措施
1. 保持向后兼容（双模式支持）
2. 充分的测试覆盖
3. 逐步迁移，不强制切换
4. 保留代码备份

## 总结

这次重构的核心目标是：
1. **统一** - 统一接口、命名、导出方式
2. **简化** - 删除冗余代码和缓存逻辑
3. **配置驱动** - 使用 voiceModels.json 作为唯一真相源
4. **向后兼容** - 不破坏现有功能

已完成的部分（配置层、核心层）为后续重构奠定了基础。剩余的服务层和路由层改动可以逐步实施。
