# 重构前代码现状分析

## 📋 概述

本文档展示需要重构的**7个TTS服务**、**旧路由文件**和**UnifiedTtsController**的当前状态（重构前），并指出需要改进的地方。

## 1️⃣ TTS服务层现状（7个服务）

### 问题总结

| 问题 | 影响 | 示例 |
|------|------|------|
| ❌ **不继承BaseTtsService** | 无法使用统一的参数验证 | 所有服务 |

| ❌ **不使用audioStorageManager** | 文件保存逻辑重复 | 所有服务 |

#### 问题清单
- ❌ 不继承 `BaseTtsService`
- ❌ 双方法名（synthesize + convertTextToSpeech）
- ❌ 不使用 `this.validateOptions()`
- ❌ 不使用 `audioStorageManager`
- ❌ 参数直接传给API：`parameters: { rate: options.rate || 1 }` (第99行)
- ❌ 硬编码路径：`const baseDir = config.paths.audioStorage || path.join(__dirname, '..', '..', '..', 'src', 'storage', 'uploads', 'audio')`

#### 需要改成
```javascript
class CosyVoiceService extends BaseTtsService {  // ✅ 继承基类
  constructor(config = {}) {
    super({
      provider: 'aliyun',
      serviceType: 'cosyvoice',
      ...config
    });
    // ...
  }

  async synthesize(text, options = {}) {  // ✅ 只保留一个方法
    // ✅ 使用基类验证
    this.validateText(text);
    const apiParams = this.validateOptions(options);  // ✅ 使用ParameterMapper

    // 调用API...
    const audioBuffer = await this.callCosyVoiceAPI(text, apiParams);

    // ✅ 使用统一存储管理
    const audioFile = await this.audioStorage.saveAudioFile(audioBuffer, {
      extension: options.format || 'mp3',
      metadata: { /* ... */ }
    });

    return { /* ... */ };
  }
}

module.exports = new CosyVoiceService();
```

## 3️⃣ UnifiedTtsController.js 现状

### 当前状态（第1-56行）

```javascript
const { ttsServiceManager } = require('./core/TtsServiceManager');
const TtsException = require('./core/TtsException');

class UnifiedTtsController {
  constructor() {
    this.serviceManager = ttsServiceManager;
  }

  async synthesize(req, res) {
    try {
      const { service, text, ...options } = req.body;

      // ✅ 验证必要参数
      if (!service || !text) {
        throw TtsException.BadRequest('Service and text parameters are required');
      }

      // ✅ 解析服务标识
      const { provider, serviceType } = this.parseServiceIdentifier(service);

      // ❌ 重复的文本验证（BaseTtsService已有）
      this.validateText(text);

      // ❌ 重复的选项验证（BaseTtsService已有）
      this.validateOptions(options);

      // ✅ 使用服务管理器
      const result = await this.serviceManager.synthesize(provider, serviceType, text, options);

      res.json({
        success: true,
        data: result,
        service: service,
        fromCache: (result && result.fromCache) || false,  // ❌ 缓存逻辑残留
        metadata: {
          provider: provider,
          serviceType: serviceType,
          requestId: req.requestId || null
        },
        timestamp: new Date().toISOString()
      });

    } catch (error) {
      this.handleError(error, res);
    }
  }

  // ❌ 需要添加 systemId 支持
  // ❌ 需要删除 fromCache 逻辑
}
```

### 问题清单

1. ❌ **重复的文本验证**（第32-35行）
   - BaseTtsService 已经有 `validateText()`
   - TtsServiceManager.synthesize() 内部会调用服务的 validateText()

2. ❌ **重复的选项验证**（第34-35行）
   - 服务内部会调用 `validateOptions()`
   - ParameterMapper 已经验证过了

3. ❌ **缓存逻辑残留**（第44行）
   - `fromCache: (result && result.fromCache) || false`
   - 缓存已被删除，这个字段没有意义

4. ❌ **不支持 systemId**
   - 需要添加对 `systemId` 参数的支持
   - 使用 `voiceModelMapper.systemIdToProviderParams()`

5. ⚠️ **缺少clearCache()方法**
   - 如果存在，需要删除

#### 需要改成
```javascript
const { ttsServiceManager } = require('./core/TtsServiceManager');
const { voiceModelMapper } = require('./config/VoiceModelMapper');
const TtsException = require('./core/TtsException');

class UnifiedTtsController {
  constructor() {
    this.serviceManager = ttsServiceManager;
  }

  async synthesize(req, res) {
    try {
      const { service, systemId, text, ...options } = req.body;

      // ✅ 支持新旧两种方式
      let provider, serviceType, finalOptions;

      if (systemId) {
        // ✅ 新方式：使用 systemId
        const params = voiceModelMapper.systemIdToProviderParams(systemId);
        provider = params.provider;
        serviceType = params.service;
        finalOptions = { ...params, ...options };  // 合并参数
      } else {
        // ✅ 旧方式：使用 service 参数
        if (!service || !text) {
          throw TtsException.BadRequest('Service and text parameters are required');
        }
        const parsed = this.parseServiceIdentifier(service);
        provider = parsed.provider;
        serviceType = parsed.serviceType;
        finalOptions = options;
      }

      // ✅ 验证文本（委托给服务层）
      if (!text) {
        throw TtsException.TextValidationError('Text is required');
      }

      // ✅ 调用服务管理器（内部会验证参数）
      const result = await this.serviceManager.synthesize(
        provider,
        serviceType,
        text,
        finalOptions
      );

      res.json({
        success: true,
        data: result,
        service: service || systemId,
        metadata: {
          provider: provider,
          serviceType: serviceType,
          requestId: req.requestId || null
        },
        timestamp: new Date().toISOString()
      });

    } catch (error) {
      this.handleError(error, res);
    }
  }

  // ❌ 删除 clearCache() 方法（如果存在）
  // ❌ 删除 validateText() 方法（重复）
  // ❌ 删除 validateOptions() 方法（重复）
}
```
