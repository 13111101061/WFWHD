### 3. 基类集成

#### BaseTtsService.js
**位置:** `src/modules/tts/core/BaseTtsService.js`

**修改内容 (第77-100行):**
```javascript
validateOptions(options = {}) {
  const { parameterMapper } = require('../config/ParameterMapper');

  // 确保ParameterMapper已初始化
  if (!parameterMapper.loaded) {
    parameterMapper.initialize();
  }

  // 使用ParameterMapper进行验证和映射
  try {
    const apiParams = parameterMapper.mapAndValidate(
      this.provider,
      this.serviceType,
      options
    );
    return apiParams;
  } catch (error) {
    // 回退到基础验证
    if (error.code === 'NOT_FOUND') {
      return this.fallbackValidation(options);
    }
    throw error;
  }
}
```

**集成特性:**
- ✅ 自动初始化 ParameterMapper
- ✅ 调用配置驱动的映射和验证
- ✅ 错误回退机制（向后兼容）
- ✅ 返回映射后的 API 参数

## 🎯 解决的核心问题

### Before (重构前)
```javascript
// ❌ 每个服务手动转换参数
cosyVoiceService.js:  rate = options.speed  // 分散逻辑
tencentTtsService.js:  Speed = options.speed
minimaxTtsService.js:  voice_setting.speed = options.speed

// ❌ 硬编码的范围验证
if (speed < 0 || speed > 5.0) throw Error('Speed invalid');

// ❌ 不一致的标准化处理
volume = options.volume * 10;  // Aliyun
volume = options.volume / 10;  // MiniMax
volume = options.volume;       // Tencent

// ❌ 混乱的错误处理
throw new Error('Invalid speed');
throw TtsException.BadRequest('Invalid volume');
return res.status(400).json({error: 'Invalid pitch'});
```

### After (重构后)
```javascript
// ✅ 统一的参数映射（配置驱动）
const apiParams = parameterMapper.mapAndValidate(
  this.provider,
  this.serviceType,
  options
);

// ✅ 配置文件定义所有规则
ProviderConfig.json:
{
  "speed": {
    "apiField": "rate",  // 或 "Speed", "voice_setting.speed"
    "range": {"min": 0.5, "max": 2.0},
    "standardization": {"transform": "value * 10"}
  }
}

// ✅ 统一的错误处理
throw TtsException.BadRequest('speed 必须在 0.5 到 2.0 之间');

// ✅ 自动化的标准化转换
// 8 → 80 (Aliyun)
// 8 → 0.8 (MiniMax)
// 8 → 8 (Tencent)
```
