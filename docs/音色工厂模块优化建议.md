### 1. 音色模型数据存储优化

**问题**: 当前使用内存Map存储音色模型数据，当模型数量达到1000+时可能存在性能问题

**当前实现**:
```javascript
// VoiceModelRegistry.js
this.models = new Map();        // 主存储
this.providers = new Map();     // 提供商索引
this.categories = new Map();    // 分类索引
this.tags = new Map();          // 标签索引
```

**优化方案**:

#### 方案A: 引入Redis缓存层
```javascript
// 新增: RedisVoiceModelCache.js
class RedisVoiceModelCache {
  async loadModels() {
    const cached = await redis.get('voice_models');
    if (cached) {
      return JSON.parse(cached);
    }
    const models = await this.loadFromConfig();
    await redis.setex('voice_models', 3600, JSON.stringify(models));
    return models;
  }
}
```

**优点**:
- 支持大规模数据存储（10000+模型）
- 支持分布式部署
- 持久化存储

**缺点**:
- 增加运维成本
- 需要额外的Redis服务

#### 方案B: 优化内存索引结构
```javascript
// 改进: VoiceModelRegistry.js
class OptimizedVoiceModelRegistry {
  constructor() {
    this.models = new Map();
    
    // 使用复合索引提升查询性能
    this.compositeIndex = {
      provider_category: new Map(),  // provider_category -> Set<modelId>
      provider_gender: new Map(),    // provider_gender -> Set<modelId>
      category_language: new Map(),  // category_language -> Set<modelId>
    };
    
    // 添加全文搜索索引
    this.searchIndex = this.buildSearchIndex();
  }
}
```

**优点**:
- 无需额外依赖
- 查询性能提升
- 实现简单

**推荐**: 先采用方案B，如果数据量持续增长再考虑方案A

### 2. API响应优化

**问题**: 当前所有API都返回完整的模型数据，包含大量可选字段

**当前实现**:
```javascript
// voiceRoutes.js
router.get('/models', async (req, res) => {
  const models = voiceModelRegistry.getAllModels();
  res.json({
    success: true,
    data: models,  // 返回完整数据
    count: models.length
  });
});
```

**优化方案**:

#### 添加字段过滤功能
```javascript
router.get('/models', async (req, res) => {
  const { fields, limit, offset } = req.query;
  let models = voiceModelRegistry.getAllModels();
  
  // 字段过滤
  if (fields) {
    const fieldList = fields.split(',');
    models = models.map(model => 
      fieldList.reduce((acc, field) => {
        if (model[field] !== undefined) {
          acc[field] = model[field];
        }
        return acc;
      }, {})
    );
  }
  
  // 分页
  if (limit) {
    const start = parseInt(offset) || 0;
    const end = start + parseInt(limit);
    models = models.slice(start, end);
  }
  
  res.json({
    success: true,
    data: models,
    count: models.length,
    total: voiceModelRegistry.getStats().totalModels
  });
});
```

**使用示例**:
```
GET /api/tts/voices/models?fields=id,name,provider,category&limit=20&offset=0
```

### 3. 缓存策略优化

**问题**: TtsServiceManager中的缓存功能被禁用

**当前实现**:

```javascript
// TtsServiceManager.js
// 暂时禁用缓存系统，直接生成音频

```

**优化方案**:

#### 重新启用并优化缓存
```javascript
class OptimizedTtsServiceManager {
  async synthesize(provider, serviceType, text, options = {}) {
    const cacheKey = generateCacheKey(text, serviceKey, options);
    
    // 检查缓存
    if (hasCache(cacheKey)) {
      const cachedData = getCache(cacheKey);
      this.metrics.cacheHits++;
      return { ...cachedData, fromCache: true };
    }
    
    // 执行合成
    const result = await ttsService.synthesize(text, options);
    
    // 异步缓存结果
    setCache(cacheKey, result, 300).catch(err => {
      console.error('缓存保存失败:', err);
    });
    
    return result;
  }
}
```

- 音色列表 → 缓存1小时
- 统计信息 → 缓存5分钟

### 4. 服务实例池化

**问题**: TtsFactory每次都创建新的服务实例

**优化方案**:

#### 实现服务实例池
```javascript
class ServicePool {
  constructor(maxSize = 10) {
    this.pool = new Map();
    this.maxSize = maxSize;
    this.usageCount = new Map();
  }
  
  acquire(serviceKey) {
    if (!this.pool.has(serviceKey)) {
      if (this.pool.size >= this.maxSize) {
        this.evictLeastUsed();
      }
      this.pool.set(serviceKey, this.createService(serviceKey));
    }
    this.usageCount.set(serviceKey, (this.usageCount.get(serviceKey) || 0) + 1);
    return this.pool.get(serviceKey);
  }
  
  evictLeastUsed() {
    const leastUsed = Array.from(this.usageCount.entries())
      .sort((a, b) => a[1] - b[1])[0];
    this.pool.delete(leastUsed[0]);
    this.usageCount.delete(leastUsed[0]);
  }
}
```

### 2. 实时流式合成

**实现方案**:
```javascript
// 新增: services/StreamingTtsService.js
class StreamingTtsService extends BaseTtsService {
  async synthesizeStream(text, options = {}) {
    const ws = new WebSocket(this.getStreamingEndpoint(options));
    
    return new Promise((resolve, reject) => {
      const audioChunks = [];
      
      ws.on('message', (data) => {
        if (this.isAudioChunk(data)) {
          audioChunks.push(data);
          // 立即发送给客户端
          if (options.onChunk) {
            options.onChunk(data);
          }
        }
      });
      
      ws.on('close', () => {
        const audioBuffer = Buffer.concat(audioChunks);
        resolve({ audioBuffer, audioChunks });
      });
      
      ws.on('error', reject);
      
      // 发送文本
      ws.send(JSON.stringify({ text, ...options }));
    });
  }
}
```

### 3. 音色推荐系统

**需求**: 根据文本内容推荐合适的音色

**实现方案**:
```javascript
// 新增: services/VoiceRecommender.js
class VoiceRecommender {
  constructor(registry) {
    this.registry = registry;
    this.rules = [
      this.createEmotionRule(),
      this.createGenderRule(),
      this.createLanguageRule(),
      this.createStyleRule()
    ];
  }
  
  recommend(text, options = {}) {
    const analysis = this.analyzeText(text);
    const allModels = this.registry.getAllModels();
    
    let scores = allModels.map(model => ({
      model,
      score: this.calculateScore(model, analysis, options)
    }));
    
    scores.sort((a, b) => b.score - a.score);
    
    return scores.slice(0, options.limit || 5).map(s => s.model);
  }
  
  analyzeText(text) {
    return {
      length: text.length,
      language: this.detectLanguage(text),
      emotion: this.detectEmotion(text),
      keywords: this.extractKeywords(text)
    };
  }
  
  calculateScore(model, analysis, options) {
    let score = 0;
    
    // 语言匹配
    if (model.languages.includes(analysis.language)) {
      score += 30;
    }
    
    // 情感匹配
    if (this.emotionMatches(model, analysis.emotion)) {
      score += 20;
    }
    
    // 关键词匹配
    const keywordMatches = model.tags.filter(tag => 
      analysis.keywords.some(kw => kw.includes(tag))
    ).length;
    score += keywordMatches * 10;
    
    // 热门度
    score += (model.popularity || 0) * 0.5;
    
    return score;
  }
}
```

```javascript
// 新增: security/InputSanitizer.js
class InputSanitizer {
  static sanitizeText(text) {
    if (!text || typeof text !== 'string') {
      throw new Error('Invalid text input');
    }
    
    // 移除控制字符
    text = text.replace(/[\x00-\x1F\x7F]/g, '');
    
    // 限制长度
    if (text.length > 10000) {
      throw new Error('Text too long');
    }
    
    // 检查是否包含可疑内容
    const suspiciousPatterns = [
      /<script/i,
      /javascript:/i,
      /on\w+\s*=/i
    ];
    
    for (const pattern of suspiciousPatterns) {
      if (pattern.test(text)) {
        throw new Error('Suspicious content detected');
      }
    }
```

```javascript
class TtsMonitor {
  constructor() {
    this.metrics = {
      requests: new Map(),
      errors: new Map(),
      latency: new Map()
    };
    
    this.alerts = {
      highErrorRate: 0.1,      // 10%错误率告警

      lowSuccessRate: 0.9      // 90%成功率告警
    };
  }
  
  recordRequest(service, success, latency) {
    const now = Date.now();
    
    if (!this.metrics.requests.has(service)) {
      this.metrics.requests.set(service, []);
    }
    
    this.metrics.requests.get(service).push({ time: now, success, latency });
    
    // 检查告警条件
    this.checkAlerts(service);
  }
  
  checkAlerts(service) {
    const requests = this.metrics.requests.get(service);
    const recent = this.getRecentRequests(requests, 60000); // 最近1分钟
    
    if (recent.length < 10) return;
    
    const errorRate = recent.filter(r => !r.success).length / recent.length;
    const avgLatency = recent.reduce((sum, r) => sum + r.latency, 0) / recent.length;
    
    if (errorRate > this.alerts.highErrorRate) {
      this.sendAlert(service, `High error rate: ${(errorRate * 100).toFixed(2)}%`);
    }
    
    if (avgLatency > this.alerts.highLatency) {
      this.sendAlert(service, `High latency: ${avgLatency.toFixed(0)}ms`);
    }
  }
  
  sendAlert(service, message) {
    console.error(`🚨 ALERT [${service}]: ${message}`);
    // 发送到告警系统（钉钉、企业微信、邮件等）
  }
```

```javascript
// 新增: monitoring/PerformanceProfiler.js
class PerformanceProfiler {
  constructor() {
    this.profiles = new Map();
  }
  
  startProfile(serviceKey) {
    if (!this.profiles.has(serviceKey)) {
      this.profiles.set(serviceKey, {
        samples: [],
        startTime: Date.now()
      });
    }
    
    return {
      serviceKey,
      start: Date.now()
    };
  }
  
  endProfile(profile) {
    const duration = Date.now() - profile.start;
    const serviceProfile = this.profiles.get(profile.serviceKey);
    
    serviceProfile.samples.push(duration);
    
    return this.getStatistics(serviceProfile.samples);
  }
  
  getStatistics(samples) {
    const sorted = [...samples].sort((a, b) => a - b);
    const sum = samples.reduce((a, b) => a + b, 0);
    
    return {
      count: samples.length,
      min: sorted[0],
      max: sorted[sorted.length - 1],
      avg: sum / samples.length,
      p50: sorted[Math.floor(sorted.length * 0.5)],
      p95: sorted[Math.floor(sorted.length * 0.95)],
      p99: sorted[Math.floor(sorted.length * 0.99)]
    };
  }
}
```
