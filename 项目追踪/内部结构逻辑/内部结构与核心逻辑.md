## DATA_CONTRACTS
- request_contract:
  required:
    - text (string, max 5000 chars)
  optional:
    - service (string, e.g. "moss_tts", "aliyun_qwen_http")
    - voice (string, system voice ID or provider real ID)
    - voiceId (string, alias for voice)
    - options (object, { speed, pitch, volume, format, ... })

- resolved_contract:
  fields:
    - providerKey (e.g. "moss")
    - serviceKey (e.g. "tts")
    - adapterKey (e.g. "moss_tts")
    - voiceId (e.g. "moss-tts-ashui")
    - voiceRuntime (object with voiceId, model, providerOptions)
    - runtimeOptions (merged: defaults + runtime + user, voice LOCKED from runtime)

- adapter_result_contract:
  variants:
    - remote_url: { url, format, size, isRemote: true, ...metadata }
    - local_audio: { audio: Buffer, format, filePath, isRemote: false, ...metadata }

- response_contract:
  success:
    - success: true
    - data: { text, audioUrl, provider, serviceType, voice, format, fileSize, sampleRate, duration, ... }
    - service: adapterKey
    - fromCache: boolean
    - metadata: { provider, serviceType, systemId, requestId }
    - timestamp: ISO string
  error:
    - success: false
    - error: message
    - code: error code (see error codes below)
    - timestamp: ISO string

## STATE_MACHINE_CREDENTIAL_HEALTH
- states:
  - healthy (0 consecutive failures)
  - degraded (1-4 consecutive failures)
  - unhealthy (5+ consecutive failures, circuit open)
- transitions:
  - success -> healthy, reset failure count
  - failure -> increment failure count
  - failures >= 5 -> circuit_open, disable account
  - after 60s -> half-open probing (single test call)
  - probe success -> healthy
  - probe fail -> remain circuit_open
