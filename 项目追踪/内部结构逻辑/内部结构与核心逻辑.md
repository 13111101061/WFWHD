## STATE_MACHINE_CREDENTIAL_HEALTH
- states:
  - healthy
  - degraded
  - unhealthy
  - circuit_open
- transitions:
  - failure_count >= threshold -> circuit_open
  - reset_timeout reached -> half-open probing path
  - probe success -> healthy

## FAILURE_TRIAGE_MATRIX
- unknown_service:
  code: UNKNOWN_SERVICE
  first_check: VoiceResolver.resolve
- provider_not_configured:
  code: PROVIDER_NOT_CONFIGURED
  first_check: credentials.isConfigured gate
- timeout:
  code: TIMEOUT_ERROR
  first_check: withTimeout + provider latency
- adapter_instantiation_error:
  code: CONFIG_ERROR or runtime throw
  first_check: adapterKey canonical mapping
